Files CCore/inc/algon/ApplyToList.h CCore/src/algon/ApplyToList.cpp

ApplyToList() applies a functor to the list of arguments. A result is returned using the GetResult(). 


template <class FuncInit,class ... TT>
auto Algon::ApplyToList(FuncInit func_init,TT && ... tt)
 {
  FunctorTypeOf<FuncInit> func(func_init);

  ( (void)func( std::forward<TT>(tt) ) , ... );

  return GetResult(func);
 }
