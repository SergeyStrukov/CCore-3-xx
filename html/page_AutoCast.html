AutoCast helps implicit pointer conversion. 


template <class T>
class AutoCast
 {
   T *ptr;

  public:

   explicit AutoCast(T *ptr_) : ptr(ptr_) {}

   template <class S>
   operator S * () const { return static_cast<S *>(ptr); }
 };

It can be used for implicit pointer conversion: 


T *ptr = .... ;

S *dst = AutoCast(ptr) ; // requires static_cast<S *>(T *)
