This Class-function is to obtain the full path of the running executable. 


#ifndef CCore_inc_sys_SysLaunchPath_h
#define CCore_inc_sys_SysLaunchPath_h

#include <CCore/inc/GenFile.h>

#include <CCore/inc/sys/SysError.h>

namespace CCore {
namespace Sys {

/* classes */

struct LaunchPath;

/* struct LaunchPath */

struct LaunchPath
 {
  StrLen path;
  ErrorType error;

  explicit LaunchPath(char buf[MaxPathLen+1]);
 };

} // namespace Sys
} // namespace CCore

#endif

To call the function you must provide a buffer with the proper length. The path is copied to this buffer. path is the resulting path, error is the error code.
