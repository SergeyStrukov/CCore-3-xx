PrintTime is a Helper Type, it prints a time, represented by an unsigned integral type in seconds. 


template <UIntType UInt>
class PrintTime
 {
   UInt value;
   
  public: 
  
   explicit PrintTime(UInt value_) : value(value_) {}
  
   void print(PrinterType &out) const;
 };

Time is printed in the form hh:mm:ss or dd:hh:mm:ss. 


Printf(Con,"#;\n",PrintTime(1000u)); // 00:16:40
